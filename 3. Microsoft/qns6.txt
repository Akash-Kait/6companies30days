class Solution
{
    public:
    vector<vector<char>> vec = {{'a', 'b', 'c'}, {'d', 'e', 'f'}, {'g', 'h', 'i'}, {'j', 'k', 'l'}, 
    {'m', 'n', 'o'}, {'p', 'q', 'r', 's'}, {'t', 'u', 'v'}, {'w', 'x', 'y', 'z'}};
    
    void helper(int a[], int N, int i, vector<string> &ans, string word) {
        if(i == N) {
            ans.push_back(word);
            return ;
        }
        
        for(int j = 0 ; j < vec[a[i]-2].size() ; j++) {
            helper(a, N, i + 1, ans, word + vec[a[i]-2][j]);
        }
    }
    //Function to find list of all words possible by pressing given numbers.
    vector<string> possibleWords(int a[], int N)
    {
        //Your code here
        vector<string> ans;
        string word;
        helper(a, N, 0, ans, word);
        
        return ans;
    }
};
